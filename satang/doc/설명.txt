ajax(비동기통신)
==> 서버에 요청을 한다는 것은 웹문서를 요청해서
	지금 보고있는 페이지를 요청한 페이지로 바꾸는 것이 된다.
	그런데 가끔은 전체를 바꾸는 것이 아니고
	페이지 중 일부만 변경해야 하는 경우가 있다.
	
	예를 들면 회원가입의 경우 아이디 체크 기능은
	서버에게 아이디를 전송하고 그 아이디를 사용할 수 있는지 여부만 받으면 된다.
	이 결과를 받아서 현재 작업중인 페이지에서 내용을 변경해서 사용하게 된다.
	
	이 경우처럼 페이지 전체가 요청이 돼서 다시 리로드 되는 요청을 동기통신이라고 한다.
	(==> 서버가 보내준 내용과 현재 보고있는 내용이 같게된다.)
	서버가 보내준 내용과 다른 페이지를 보고있는 경우 이런 통신을 비동기통신이라고 한다.
	
	jQuery에서는 비동기통신을 처리할 함수를 제공해주고 있는데
	그 함수가
	 $.ajax({}) 함수이다.
	 
	 형식 ]
	 	$.ajax({
	 		url:'';							==> 요청할 서버의 주소
	 		type:'';						==> 데이터 전송방식 (get || post)
	 		dataType:'';					==> 응답문서 형식
	 		data:{
	 			파라미터키값1 : 파라미터밸류1, 
	 			파라미터키값2 : 파라미터밸류2, 
	 			...
	 			
	 			예] id,pw 전송
	 				id : 'jennie',
	 				pw : '12345'	
	 		},
	 		success: function(변수이름){
	 			==> 통신에 성공한 경우 실행되는 함수
	 				매개변수는 응답문서를 기억할 변수
	 				
	 				만약 서버에서 
	 				{
	 					"result" : "OK"
	 				}
	 				라는 응답문서를 전송한 경우
	 				
	 				변수이름 = {
	 					"result" : "OK"
	 				}
	 				
	 				따라서 처리작업은 거의 매개변수에 접근해서 처리해주면된다.
	 		},
	 		error: function(){
	 			통신에 실패한 경우 실행되는 함수 
	 		}
	 	});
